import { useState, useEffect, useRef } from "react";

import { GiHamburgerMenu } from "react-icons/gi";
import { GrFormClose } from "react-icons/gr";
import { CgChevronLeftO, CgChevronRightO } from "react-icons/cg";

import Img1 from './assets/img-1.png';
import Img2 from './assets/img-2.png';
import Img3 from './assets/img-3.png';

import './App.scss';

const App = () => {

  const [ block, setBlock ] = useState([]);

  const addBlock = () => {
    const blockItem = {
      id: crypto.randomUUID(),
      text: {
        text1: 'text one',
        text2: 'text two',
        text3: 'text three'
      },
      links: [
        "https://co1.com",
        "https://co2.com",
        "https://co3.com",
      ],
      images: [
        Img1,
        Img2,
        Img3
      ]
    }

    setBlock(() => {
      const storageItem = [...block, blockItem];
      localStorage.setItem('show-block', JSON.stringify(storageItem));
      return storageItem;
    });
  }

  useEffect(()=> {
    const loadByStorage = localStorage.getItem('show-block');
    if(loadByStorage) setBlock(JSON.parse(loadByStorage));
  }, [setBlock]);

  const removeBlock = () => {
    localStorage.removeItem('show-block');
    setBlock([]);
  }

  return (
   <div className='bg-white'>
    <button type="button" className='btn btn-primary' onClick={addBlock}>Click me</button>
    {block.map((item) => (
      <div key={item.id} className='block-wrapper d-flex flex-column gap-3 align-items-center'>
        {Object.entries(item.text).map(([key, txt], index) => (
          <p key={index}><strong>{key}</strong> - {txt}</p>
        ))}
        {item.links.map((lnk, index) => (
          <a key={index} href="#" className="link">{lnk}</a>
        ))}
        {item.images.map((img, index) => (
          <img key={index} src={img} alt="Image" className="hero-img w-25 rounded-2" />
        ))}
      </div>
    ))}
    {block.length > 0 && <button type="button" className='btn btn-warning mt-2' onClick={removeBlock}>Click me</button>}
   </div>
 );
};

export default App;

import { useState, useEffect, useRef } from "react";

import { GiHamburgerMenu } from "react-icons/gi";
import { GrFormClose } from "react-icons/gr";
import { CgChevronLeftO, CgChevronRightO } from "react-icons/cg";

import Img1 from './assets/img-1.png';
import Img2 from './assets/img-2.png';
import Img3 from './assets/img-3.png';

import './App.scss';

const App = () => {

  const [ isOpen, setIsOpen ] = useState(false);

  const toggleMenu = () => {
    setIsOpen(prev => !prev);
    isOpen ? localStorage.removeItem('menu') : localStorage.setItem('menu', 'true');
  }

  useEffect(()=> {
    const loadMenuByStorage = localStorage.getItem('menu');
    if(loadMenuByStorage) setIsOpen(true);
  }, [isOpen]);


  return (
   <div className='bg-white'>
    <button type="button" className='btn btn-primary' onClick={toggleMenu}>
      {isOpen ? <GrFormClose /> : <GiHamburgerMenu />}
    </button>
    <div className={`menu pt-2 ${isOpen ? 'd-block' : 'd-none'}`}>
      <a href="#" className="link">Home</a>
      <a href="#" className="link">About</a>
      <a href="#" className="link">Contact</a>
    </div>
   </div>
 );
};

export default App;

const ThemeProvider = ({ children }) => {

    const [ theme, setTheme ] = useState(()=> {
        return localStorage.getItem('theme') || 'light';
    });

    useEffect(()=> {
        document.body.classList.remove('light', 'dark');
        document.body.classList.add(theme);

        if(theme === 'light') {
        document.body.style.backgroundColor = '#fff';
        document.body.style.color = '#333';
        } else {
            document.body.style.backgroundColor = '#333';
            document.body.style.color = '#fff';
        }

        localStorage.setItem('theme', theme);
    }, [theme]);

    const toggleTheme = () => {
        setTheme((prev) => prev === 'light' ? 'dark' : 'light');
    }

  return (
   <ThemeContext.Provider value={{theme, toggleTheme}}>
        { children }
   </ThemeContext.Provider>
 );
};

export function useCartQuantity(setCart) {

    const increment = (id) => {
        setCart(prev =>
                prev.map(item =>
                item.id === id
                ? {...item, quantity: Math.min(10, item.quantity +1)}
                : item
                )
        )
    }

    const decrement = (id) => {
        setCart(prev =>
                prev.map(item =>
                    item.id === id
                    ? {...item, quantity: Math.max(1, item.quantity -1)}
                    : item
                )
        )
    }

    return { increment, decrement };
};

import { useState, useEffect } from 'react'
export function useFetch(url) {
    const [ data, setData ] = useState([]);
    const [loading, setLoading] = useState(!!url);
    const [ error, setError ] = useState(null);
    useEffect(()=> {
      const fetchData = async () => {
        let isMount = true;
        try {
            const request = await fetch(url);
            if(!request.ok) throw new Error('Request failed');
            const response = await request.json();
            if(isMount) setData(response);
        } catch (error) {
            if(isMount) setError(error.message);
        } finally {
            if(isMount) setLoading(false);
        }
      }
      fetchData();
      return () => {
        isMount = false;
      }
    }, [url]);
    return { data, loading, error };
};
